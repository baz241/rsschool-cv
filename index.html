              const maxSubarray = (nums, k) => {
                let left = 0;
                let right = 0;
                let max;
                let sum = 0;
                for (right; right < nums.length; right++) {
                    sum += nums[right];
                    if (right - left === k - 1) {
                        if (max === undefined) {
                            max = sum;
                        } else if (sum > max) {
                            max = sum;
                        }
                        sum -= nums[left];
                        left += 1;
                    }
                }
                return max / k;
              }
